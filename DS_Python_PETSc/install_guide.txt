	1. Download PETSc

	git clone -b release https://gitlab.com/petsc/petsc.git petsc
	
	2. Setup the path for PETSc

	export PETSC_DIR=/home/liweiw/petsc4py/code/DS_Petsc/petsc
	export PETSC_ARCH=arch-linux-c-debug
	
	3. Load dependencies

	pip3 install --user Cython
	openmpi/3.1.6-gcc/8.3.1-cuda10_2-ucx
	
	4. Install and make

	./configure --download-fblaslapack --with-mpi-dir=/software/spackages/linux-centos8-x86_64/gcc-8.3.1/openmpi-3.1.6-otcopqkdim5tm5ndifcmjlmzqpwcm2xv --with-scalar-type=complex --with-precision=double --download-petsc4py --download-make --download-superlu_dist
	
	make PETSC_DIR=/home/liweiw/petsc4py/code/DS_Petsc/petsc PETSC_ARCH=arch-linux-c-debug all
	
	make PETSC_DIR=/home/liweiw/petsc4py/code/DS_Petsc/petsc PETSC_ARCH=arch-linux-c-debug check

	5. Add petsc4py lib to python path
	
	export PYTHONPATH="${PYTHONPATH}:/home/liweiw/petsc4py/code/DS_Petsc/petsc/arch-linux-c-debug/lib"
	
	
Another optional packages: --download-superlu --download-scalapack --download-mumps --download-metis --download-parmetis
